with open(config_file) as f:
    params = json.load(f)

img_path = params["img_path"]
outdir = params["outdir"]
name = params["name"]
step_size = params["step_size"] if "step_size" in params else 1.0
step_width = params["step_width"] if "step_width" in params else 1.0
alpha = params["alpha"] if "alpha" in params else 1.0
beta = params["beta"] if "beta" in params else 1e-3
friction = params["friction"] if "friction" in params else 1e-4
repeat_starts = params["repeat_starts"] if "repeat_starts" in params else False
patch_radius = 17
inchannels = 1

if "classifier_weights" in params:
    classifier_path = params["classifier_weights"]
    classifier_state_dict = torch.load(classifier_path)#, weights_only=True)
    classifier = ResNet3D(ResidualBlock3D, [3, 4, 6, 3], in_channels=inchannels, num_classes=1)
    classifier = classifier.to(device=DEVICE, dtype=dtype)
    classifier.load_state_dict(classifier_state_dict)
    classifier.eval()
else:
    classifier = None

env = Environment(img_path,
                radius=patch_radius,
                step_size=step_size,
                step_width=step_width,
                max_len=10000,
                max_paths=1000,
                alpha=alpha,
                beta=beta,
                friction=friction,
                classifier=classifier,
                repeat_starts=repeat_starts)

import os
import numpy as np

outdir = params["outdir"]
out_files = os.listdir(outdir)
max_coverages = []
coverage_max_returns = []

for out in out_files:
    out = os.path.join(outdir, out)
    inference_outputs = np.load(out, allow_pickle=True)
    coverages = inference_outputs["coverages"]
    estimated_returns = inference_outputs["estimated_returns"]
    max_coverages.append(np.max(coverages))
    coverage_max_returns.append(coverages[np.argmax(estimated_returns)])
